298  def input_formatted_text(self, inpt):
299  """
300  Input text formatted using TeX / LateX, setting the right tags.
301  The text is first parsed through a special parser, that
302  deciphers all the commands used and translates to tags.
303 
304  @see tex_parser.py
305  """
306  self.delete(1.0, END)
307  formatted = tex_parse(inpt)
308  # print(formatted)
309  if not formatted: return
310 
311  last_position = self.index(INSERT)
312  current_tags = set()
313  for f, tags in formatted:
314  self.insert(INSERT, f)
315  # print(last_position)
316  if set(tags) != current_tags:
317  for tag in tags:
318  self.tag_add(tag, last_position, INSERT)
319  last_position = self.index(INSERT)

321  def input_tex_with_comments(self, inpt):
322  lines = inpt.split("\n")
323  comments = [l for l in lines if l.startswith("%")]
324  tex = '\n'.join([l for l in lines if not l.startswith("%")])
325  self.input_formatted_text(tex)
326  try:
327  for l in comments:
328  if l.startswith("%-CURSOR-%"):
329  cursor = l[11:]
330  # print("Cursor is in position %s" % cursor)
331  self.mark_set(INSERT, cursor)
332  self.see(INSERT)
333  elif l.startswith("%-TAG-%"):
334  d = json.loads(l[7:])
335  if d["t"] in self.tag_names():
336  self.tag_add(d["t"], d["b"], d["e"])
337  else:
338  print("Warning : unknown tag name %s" % d["t"])
339  except Exception as e:
340  print("Error while processing tags")

349  def output_tex_with_comments(self, verbose=False):
350  tmp = self.get_tex_formatted_text(self)
