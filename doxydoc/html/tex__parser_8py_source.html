228 def tex_to_tags(inpt):
229  """
230  Used only for debugging / testing purposes, since the interface
231  between parsed tex and tags is normally made by the tkinter
232  text zone itself...
233  """
234  tmp = tex_parse(inpt)
235  if not tmp:
236  return []
237 
238  res = []
239  current_tags = tmp[0][1]
240  for t in current_tags:
241  res.append(('tagon', t, ''))
242  current_text = tmp[0][0]
243 
244  for text, tags in tmp[1:]:
245  if tags == current_tags:
246  current_text += text
247  else:
248  # tags have changed
249  res.append(('text', current_text, ''))
250  current_text = text
251  add_tags = set(tags) - set(current_tags)
252  remove_tags = set(current_tags) - set(tags)
253 
254  for t in add_tags:
255  res.append(('tagon', t, ''))
256  for t in remove_tags:
257  res.append(('tagoff', t, ''))
258  current_tags = tags
259  if current_text:
260  res.append(('text', current_text, ''))
261  if current_tags:
262  for t in current_tags:
263  res.append(('tagoff', t, ''))
264

287 
288  elif a == 'text':
289

292  for t in MAPPING_OUT_NORMAL[tag]:
293  res += "}"
