39 def _cut_groups(word):
40  """
41  Coupe un mot en groupes de lettres qui sont soit des voyelles,
42  soit des consonnes.
43  Grosso modo, une syllabe par groupe de voyelles.
44  """
45  l = list()
46  current = ""
47  if word == "":
48  return []
49  isvow = word[0] in VOYELLES
50  for c in word:
51  if isvow == (c in VOYELLES):
52  current += c
53  else:
54  isvow = not isvow
55  l.append(current)
56  current = c
57  l.append(current)
58  return l
