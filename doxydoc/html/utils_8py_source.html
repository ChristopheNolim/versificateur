378  def on_find(self, event=None):
379  """
380  The method has two behaviors :
381  - if some text is selected, search and find in this part only
382  - if no text is selected, search in the whole text
383  """
384  target = askstring(
385  'Chercher un motif',
386  'Sélectionnez un motif :',
387  initialvalue="")
388  if target:
389  self.tag_remove("Find", 1.0, END)
390  begin, end = None, None
391  try:
392  text = self.get(SEL_FIRST, SEL_LAST)
393  # if there is an exception here : no selection. Search in the
394  # whole text
395  begin, end = SEL_FIRST, SEL_LAST
396  except Exception as e:
397  begin, end = 1.0, END
398  where = self.search(target, begin, end, nocase=True)
399  while where:
400  pastit = '{}+{}c'.format(where, len(target))
401  pastit2 = '{}+{}c'.format(where, len(target)-1)
402  self.tag_add("Find", where, pastit)
403  begin = pastit
404  where = self.search(target, begin, end, nocase=True)
405  # search has ended
406  if not where:
407  self.mark_set(INSERT, pastit2)
